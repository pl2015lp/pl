from nltk.corpus import gutenberg
gutenberg.fileids()

stopwords=nltk.corpus.stopwords.words('people')

bigrams=nltk.bigrams(k)
fdist= nltk.FreqDist([b for b in bigrams if b[0] not in stopwords and b[1] not in stopwords])
